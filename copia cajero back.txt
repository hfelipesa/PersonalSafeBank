let registros = [];

// Obtén una referencia al botón de registro
let registro = document.getElementById("registro");

registro.addEventListener("click", () => { 
   // Obtén los valores de los campos del formulario de registro
   let nuevoUsuario = document.getElementById("nuevoUsuario").value;
   let nuevaContrasena = document.getElementById("nuevaContrasena").value;
   let confirmarContrasena = document.getElementById("confirmarContrasena").value;
   let correoElectronico = document.getElementById("Correo").value;
   let monto = document.getElementById("monto").value;

   // Realiza validación de los datos aquí si es necesario
   if (nuevoUsuario && nuevaContrasena && nuevaContrasena === confirmarContrasena && correoElectronico && monto >= 100000) {
       // Crea un objeto que representa el registro
       let nuevoRegistro = {
           usuario: nuevoUsuario,
           contrasena: nuevaContrasena,
           correo: correoElectronico,
           monto: monto
       };

       // Agrega el nuevo registro al arreglo
       registros.push(nuevoRegistro);

       // Limpia los campos del formulario
       document.getElementById("nuevoUsuario").value = "";
       document.getElementById("nuevaContrasena").value = "";
       document.getElementById("confirmarContrasena").value = "";
       document.getElementById("Correo").value = "";
       document.getElementById("monto").value = "";

       // Opcional: muestra un mensaje de éxito al usuario
       alert("Registro exitoso. Ahora puedes iniciar sesión.");
   } else {
       // Opcional: muestra un mensaje de error al usuario si la validación falla
       alert("Por favor, completa los campos correctamente.");
   }
});

function iniciarSesion(){
        if (cuentaBloqueada) {
            alert("Tu cuenta está bloqueada. Contacta al soporte.");
            return;
        }

        const usuario ="felipe"
        const contrasena ="felipe123"
        if (usuario === document.getElementById("usuario").value     && contrasena === document.getElementById("contrasena").value) {
            containerForm.style.display="none"
            containerItems.style.display="flex"
            alert("Inicio de sesión exitoso. Redirigiendo a otro menú.");
        } else {
            intentosIngreso++;
            if (intentosIngreso >= 3) {
                cuentaBloqueada = true;
                alert("Tu cuenta ha sido bloqueada debido a múltiples intentos fallidos. Contacta al soporte.");
            } else {
                alert("Credenciales incorrectas. Intento " + intentosIngreso + " de 3.");
            }
        }
    
    }
